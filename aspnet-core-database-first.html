# ASP.NET Core Database First

# Use Entity Framework to generate models

# Exemple

$connectionstring = 'server=(localdb)\mssqllocaldb;database=MyDatabase;integrated security=true;'
$provider = 'Microsoft.EntityFrameworkCore.SqlServer'
dotnet ef dbcontext scaffold $connectionstring $provider -d -o Models -c MyDatabaseContext

# Getting help

dotnet ef dbcontext scaffold -h

# Notes

  • The folder specified by the -o option will be created if it does not exist.
  • By default the context will be named after the name of the database to scaffold. Use the -c option to specify a name.
